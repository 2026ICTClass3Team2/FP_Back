CREATE TABLE user (
    user_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    nickname VARCHAR(50) NOT NULL,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL,
    password VARCHAR(255),
    role ENUM('user', 'admin') DEFAULT 'user' NOT NULL,
    status ENUM('active', 'suspended', 'deleted') NOT NULL DEFAULT 'active',
    provider ENUM('local', 'google', 'github', 'kakao') DEFAULT 'local',
    provider_id VARCHAR(255),
    profile_pic_url TEXT,
    current_point INT NOT NULL DEFAULT 0,
    follower_count INT NOT NULL DEFAULT 0,
    warning_count INT NOT NULL DEFAULT 0,
    is_suspended BOOLEAN DEFAULT FALSE NOT NULL,
    registered_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_user_nickname (nickname),
    INDEX idx_user_provider (provider, provider_id),
    INDEX idx_user_status (status),
    UNIQUE KEY uk_user_username (username),
    UNIQUE KEY uk_user_email (email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE channel (
    channel_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    channel_name VARCHAR(50) NOT NULL,
    description TEXT,
    follower_count INT DEFAULT 0 NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT,

    INDEX idx_channel_name (channel_name),
    INDEX idx_channel_count (follower_count DESC),
    CONSTRAINT fk_channel_owner FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE tag (
    tag_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    tag_name VARCHAR(50) NOT NULL,

    UNIQUE KEY uk_tag_name (tag_name)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE emote (
    emote_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    image_url TEXT NOT NULL,
    price INT NOT NULL DEFAULT 0,

    INDEX idx_emote_name (name),
    INDEX idx_emote_price (price DESC)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE post (
    post_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    thumbnail_url TEXT,
    author VARCHAR(100),
    title VARCHAR(255) NOT NULL,
    body TEXT NOT NULL,
    content_type ENUM('feed', 'qna') NOT NULL DEFAULT 'feed',
    source_type ENUM('internal', 'external') NOT NULL DEFAULT 'internal',
    status ENUM('active', 'hidden', 'frozen') NOT NULL DEFAULT 'active',
    external_id VARCHAR(100) UNIQUE,
    external_url TEXT,
    is_hidden BOOLEAN NOT NULL DEFAULT FALSE,
    is_solved BOOLEAN NOT NULL DEFAULT FALSE,
    comment_count INT NOT NULL DEFAULT 0,
    like_count INT NOT NULL DEFAULT 0,
    dislike_count INT NOT NULL DEFAULT 0,
    view_count INT NOT NULL DEFAULT 0,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    channel_id BIGINT,
    user_id BIGINT,

    FULLTEXT INDEX idx_post_search (author, title, body),
    INDEX idx_post_created (created_at DESC),
    INDEX idx_post_channel (channel_id),
    INDEX idx_post_content (content_type, created_at DESC),
    INDEX idx_post_solved (is_solved, created_at DESC),
    INDEX idx_post_like (like_count DESC, created_at DESC),
    INDEX idx_post_dislike (dislike_count DESC, created_at DESC),
    INDEX idx_post_view (view_count DESC, created_at DESC),
    CONSTRAINT fk_post_channel FOREIGN KEY (channel_id) REFERENCES channel(channel_id) ON DELETE SET NULL,
    CONSTRAINT fk_post_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE comment (
    comment_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    content TEXT NOT NULL,
    is_answer BOOLEAN NOT NULL DEFAULT FALSE,
    like_count INT NOT NULL DEFAULT 0,
    dislike_count INT NOT NULL DEFAULT 0,
    status ENUM('active', 'deleted') NOT NULL DEFAULT 'active',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    post_id BIGINT NOT NULL,
    user_id BIGINT,
    parent_id BIGINT,

    INDEX idx_comment_parent (parent_id),
    INDEX idx_comment_user (user_id),
    INDEX idx_comment_post (post_id, status, created_at DESC),
    CONSTRAINT fk_comment_post FOREIGN KEY (post_id) REFERENCES post(post_id),
    CONSTRAINT fk_comment_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE SET NULL,
    CONSTRAINT fk_comment_id FOREIGN KEY (parent_id) REFERENCES comment(comment_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE qna (
    qna_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    is_solved BOOLEAN NOT NULL DEFAULT FALSE,
    reward_points INT NOT NULL DEFAULT 0,
    answer_id BIGINT,
    post_id BIGINT NOT NULL,

    CONSTRAINT fk_qna_post FOREIGN KEY (post_id) REFERENCES post(post_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE point_transaction (
    point_transaction_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    target_id BIGINT NOT NULL,
    target_type ENUM('qna', 'emote', 'admin', 'bonus', 'comment', 'post', 'like') NOT NULL,
    point_change INT NOT NULL,
    point_balance INT NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,

    INDEX idx_point_user (user_id, created_at DESC),
    CONSTRAINT fk_point_user FOREIGN KEY (user_id) REFERENCES user(user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE interaction (
    interaction_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    target_id BIGINT NOT NULL,
    target_type ENUM('qna', 'feed', 'comments') NOT NULL,
    action_type ENUM('like', 'dislike', 'share') NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,

    INDEX idx_interaction_target (target_type, target_id),
    UNIQUE KEY uk_interaction_target (user_id, target_type, target_id),
    CONSTRAINT fk_interaction_user FOREIGN KEY (user_id) REFERENCES user(user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE chat (
    chat_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    conversation_id BIGINT NOT NULL,
    content TEXT NOT NULL,
    is_read BOOLEAN NOT NULL DEFAULT FALSE,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    sender_id BIGINT,
    receiver_id BIGINT,

    INDEX idx_chat_sender (sender_id, created_at DESC),
    INDEX idx_chat_receiver (receiver_id, created_at DESC),
    INDEX idx_chat_conversation (conversation_id, created_at DESC),
    INDEX idx_chat_read (is_read),
    CONSTRAINT fk_chat_sender FOREIGN KEY (sender_id) REFERENCES user(user_id) ON DELETE SET NULL,
    CONSTRAINT fk_chat_receiver FOREIGN KEY (receiver_id) REFERENCES user(user_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE view_history (
    view_history_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    viewed_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    post_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,

    CONSTRAINT fk_view_post FOREIGN KEY (post_id) REFERENCES post(post_id) ON DELETE CASCADE,
    CONSTRAINT fk_view_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE report (
    report_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    target_id BIGINT NOT NULL,
    target_type ENUM('post', 'comments', 'user') NOT NULL,
    category ENUM('spam', 'harrassment', 'inappropriate', 'copyright', 'other') NOT NULL,
    details TEXT NOT NULL,
    status ENUM('pending', 'approved', 'rejected') NOT NULL DEFAULT 'pending',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    reporter_id BIGINT,

    INDEX idx_report_status (status, created_at),
    INDEX idx_report_type (target_type, target_id),
    CONSTRAINT fk_report_reporter FOREIGN KEY (reporter_id) REFERENCES user(user_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE block(
    block_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    blocker_id BIGINT NOT NULL,
    blocked_id BIGINT NOT NULL,

    UNIQUE KEY uk_block_pair (blocker_id, blocked_id),
    CONSTRAINT fk_block_actor FOREIGN KEY (blocker_id) REFERENCES user(user_id),
    CONSTRAINT fk_block_target FOREIGN KEY (blocked_id) REFERENCES user(user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE suspended (
    suspended_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    reason TEXT NOT NULL,
    suspended_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    released_at DATETIME,
    user_id BIGINT NOT NULL,
    admin_id BIGINT NOT NULL,

    INDEX idx_suspended_user (suspended_id),
    CONSTRAINT fk_suspended_user FOREIGN KEY (user_id) REFERENCES user(user_id),
    CONSTRAINT fk_suspended_admin FOREIGN KEY (admin_id) REFERENCES user(user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE hidden (
    hidden_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    target_id BIGINT NOT NULL,
    target_type ENUM('feed', 'qna', 'comment') NOT NULL,
    reason ENUM('reported', 'not_interested') NOT NULL DEFAULT 'not_interested',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,

    INDEX idx_hidden_target (user_id, target_type, target_id),
    UNIQUE KEY uk_hidden_pair (user_id, target_id),
    CONSTRAINT fk_hidden_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE notification (
    notification_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    notification_type VARCHAR(50),
    target_type ENUM('post', 'user', 'channel',
    'admin', 'comment', 'report', 'mention') NOT NULL,
    target_id BIGINT NOT NULL,
    is_read BOOLEAN NOT NULL DEFAULT FALSE,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,

    INDEX idx_notification_unread (user_id, is_read, created_at DESC),
    CONSTRAINT fk_notification_receiver FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE content_tag (
    content_tag_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    post_id BIGINT NOT NULL,
    tag_id BIGINT NOT NULL,

    CONSTRAINT fk_content_post FOREIGN KEY (post_id) REFERENCES post(post_id) ON DELETE CASCADE,
    CONSTRAINT fk_content_tag FOREIGN KEY (tag_id) REFERENCES tag(tag_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE follow (
    follow_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    target_type ENUM('channel', 'user') DEFAULT 'channel' NOT NULL,
    target_id BIGINT NOT NULL,
    is_muted BOOLEAN NOT NULL DEFAULT FALSE,
    followed_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,

    INDEX idx_follow_id (target_id, target_type, is_muted, followed_at DESC),
    UNIQUE KEY uk_follow_pair (user_id, target_id, target_type),
    CONSTRAINT fk_follow_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE inventory (
    inventory_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    purchased_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,
    emote_Id BIGINT NOT NULL,

    INDEX idx_inventory (user_id, purchased_at DESC),
    CONSTRAINT fk_inventory_user FOREIGN KEY (user_id) REFERENCES user(user_id),
    CONSTRAINT fk_inventory_emote FOREIGN KEY (emote_id) REFERENCES emote(emote_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE bookmark (
    bookmark_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    target_id BIGINT NOT NULL,
    target_type ENUM('feed', 'qna') NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,

    INDEX idx_bookmark_user (user_id, target_type),
    UNIQUE KEY uk_bookmark_target (user_id, target_type, target_id),
    CONSTRAINT fk_bookmark_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE suggestion (
    suggestion_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    category VARCHAR(50),
    title VARCHAR(255),
    details TEXT,
    user_id BIGINT,

    CONSTRAINT fk_suggest_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE interest (
    interest_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    weight_score DOUBLE NOT NULL DEFAULT 1.0,
    is_profile_tag BOOLEAN NOT NULL DEFAULT FALSE,
    last_interaction_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    user_id BIGINT NOT NULL,
    tag_id BIGINT NOT NULL,

    INDEX idx_interest_weight (user_id, weight_score DESC),
    CONSTRAINT fk_interest_user FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE,
    CONSTRAINT fk_interest_tag FOREIGN KEY (tag_id) REFERENCES tag(tag_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
